Group reduAll = [Select Id from Group where DeveloperName = 'REDU_All'];
List<Group> reduGroups = [Select Id from Group where DeveloperName like 'REDU%' 
                            and DeveloperName != 'REDU_All'];

List<GroupMember> gms = new List<GroupMember>();

for (Group g : reduGroups) {
    GroupMember gm = new GroupMember();
    gm.GroupId = reduAll.Id;
    gm.UserOrGroupId = g.Id;
    gms.add(gm);
}

insert gms;
